Agrupación de datos

 
						import pandas as pd
data = {
    'Nombre': ['Ana', 'Luis', 'Carmen', 'Jose'],
    'Edad': [25, 30, 22, 27],
    'Ciudad': ['Madrid', 'Barcelona', 'Madrid', 'Valencia'],
    'Salario': [2000, 2500, 1800, 2200]
}
aggregated_data = (
    pd.DataFrame(data)  # Creación del DataFrame
    .groupby('Ciudad')  # Agrupación por 'Ciudad'
    .agg({
        'Edad': 'mean',  # Calcula la media de 'Edad'
        'Salario': 'sum'  # Suma los 'Salarios'
    })
)
print(aggregated_data)
